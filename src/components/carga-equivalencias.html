El combo de carrera de origen se obtiene de la siguiente query, pasandole como parametro el estado de la resolucion = 1 (habilitada), la carrera selecionada en el combo de carreras ues21 y la universidad de origen seleccionada:

SELECT DISTINCT C.ID C_ID,

C.DESCRIPCION C_DESCRIPCION, case

when IE.ID is NULL then 0

else IE.id

end as IE_ID, case

when IE.DESCRIPCION is NULL then 'SIN DATO'

else IE.descripcion

end as IE_DESCRIPCION, CX.DESCRIPCION CX_DESCRIPCION, CX.ID CX_ID

FROM UE_RESOLUCIONES_GENERALES RG JOIN UE_CARRERAS C

ON RG.ID_CARRERA_UES21 = C.ID

JOIN UE_INSTITUCIONES_EDUCATIVAS IE

ON IE.ID = RG.ID_INSTITUCION_EDUCATIVA AND IE.ID = ?

JOIN UE_CARRERAS_EXTERNAS CX

ON RG.ID_CARRERA_EXTERNA = CX.ID JOIN CG_REF_CODES ESR

ON ESR.RV_LOW_VALUE = RG.ESTADO WHERE ESR.RV_DOMAIN = 'ESTADO_RESOLUCION'

AND RG.ESTADO = ?

AND RG.TIPO_RESOLUCION = 1 AND C.ID = ?

SELECT DISTINCT M.Id M_ID, M.codigo M_CODIGO, M.Descripcion

M_DESCRIPCION, CM.Cuatrimestre CM_CUATRIMESTRE, M.CODIGO

CM_COD_MAT_ID

FROM UE21.UE_CARRERA_MATERIAS CM,

UE21.UE_CARRERA_PLANES_ESTUDIOS CP, UE21.UE_MATERIAS M

WHERE CM.Mat_Id = M.Id

AND CM.Cpe_Car_Id = CP.Car_Id

AND CM.Cpe_Pes_Id = CP.Pes_Id

AND CM.Cpe_Clase_Estudio = CP.Clase_Estudio

AND CM.Estado = 1

AND CM.CUATRIMESTRE <> 0

AND CP.HABILITADO_EQUIV = 1

AND CM.tipo_materia in (1,7)

ORDER BY CM.Cuatrimestre, M.Descripcion ;
